let slideIndex = 1;

function showSlides(n) {
  const slides = document.getElementsByClassName("slides");
  if (slides.length === 0) return;
  if (n > slides.length) { slideIndex = 1; }
  if (n < 1) { slideIndex = slides.length; }
  for (let i = 0; i < slides.length; i++) {
    slides[i].style.display = "none"; // hide all slides
  }
  slides[slideIndex - 1].style.display = "block"; // show current slide
}

function plusSlides(n) {
  showSlides(slideIndex += n);
}

  showSlides(slideIndex);
};

<script>
function countdown(id,date,label){
  const x=setInterval(()=>{
    const now=new Date().getTime();
    const distance=new Date(date).getTime()-now;
    if(distance<0){document.getElementById(id).innerHTML=`${label} has started!`; clearInterval(x); return;}
    const d=Math.floor(distance/(1000*60*60*24));
    const h=Math.floor((distance%(1000*60*60*24))/(1000*60*60));
    const m=Math.floor((distance%(1000*60*60))/(1000*60));
    const s=Math.floor((distance%(1000*60))/1000);
    document.getElementById(id).innerHTML=`${label}: ${d}d ${h}h ${m}m ${s}s`;
  },1000);
}
